With Prototype object model

    var BaseModel = Class.create({
      url: function(){
        return '/users/ID';
      }
    });

    var MyModel = Class.create(BaseModel, {
      url: function(){
        return this.ancestor.super('url') + '/somethingExtra';          // approach 1
        // return MyModel.prototype.super('url') + '/somethingExtra';   // approach 2
      }
    });

    var myInstance = new MyModel();

    alert(myInstance.ancestor === MyModel.prototype);                   // true
    alert(myInstance.ancestor.ancestor === BaseModel.prototype);        // true

    alert(myInstance.url());                    // "/users/ID/somethingExtra" with both approaches

    // The second level of descending
    var MyModel2 = Class.create(MyModel)
    var myInstance2 = new MyModel2();
    alert(myInstance2.url());                   // "/users/ID/somethingExtra/somethingExtra" with approach#1, "/users/ID/somethingExtra" with approach#2
