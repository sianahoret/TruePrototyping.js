With Backbone inheritance model

    var MyModel = Backbone.Model.extend({
      urlRoot : '/users',

      id: "ID",

      url: function(){
        return this.ancestor.super('url') + '/somethingExtra';
        // return MyModel.prototype.super('url') + '/somethingExtra';     // the alternative approach, it provides more strong behavior on all descending levels
      }
    });

    var myInstance = new MyModel();

Inherited method (implemented with 'ancestor' and 'super'):

    // on the first level of inheritance chain
    alert(myInstance.url());                 // "/users/ID/somethingExtra" with both approaches

    // on the second level of inheritance chain
    var MyModel2 = MyModel.extend({});
    var myInstance2 = new MyModel2();
    alert(myInstance2.url());                 // "/users/ID/somethingExtra/somethingExtra" with approach#1, "/users/ID/somethingExtra" with approach#2

ancestor:

    alert(myInstance.ancestor === MyModel.prototype);                   // true
    alert(myInstance.ancestor.ancestor === Backbone.Model.prototype);   // true

ancestors:

    myInstance.ancestors;                                               // [ MyModel.prototype, Backbone.Model.prototype, Object.prototype ]
    alert(myInstance.ancestors[0] === MyModel.prototype);               // true
    alert(myInstance.ancestors[1] === Backbone.Model.prototype);        // true
    alert(myInstance.ancestors[2] === Object.prototype);                // true

isAncestorOf:

    alert(MyModel.prototype.isAncestorOf(myInstance));                  // true
    alert(Backbone.Model.prototype.isAncestorOf(myInstance));           // true

isDescendantOf:

    alert(myInstance.isDescendantOf(MyModel.prototype));                // true
    alert(myInstance.isDescendantOf(Backbone.Model.prototype));         // true
