With Backbone

    var MyModel = Backbone.Model.extend({
      urlRoot : '/users',

      id: "ID",

      url: function(){
        return MyModel.prototype.super('url') + '/somethingExtra';
        // return this.ancestor.super('url') + '/somethingExtra';         // this approach is also possible
      }
    });

    var myInstance = new MyModel();

    alert(myInstance.ancestor === MyModel.prototype);                     // true
    alert(myInstance.ancestor.ancestor === Backbone.Model.prototype);     // true

    alert(myInstance.url());                                              // "/users/ID/somethingExtra"
