TruePrototyping

    var BaseModel = {
        url: function(){
          return '/users/ID';
        }
    };

    var MyModel = BaseModel.derive({
      url: function(){
        return this.ancestor.super('url') + '/somethingExtra';
      }
    });

    var myInstance = MyModel.derive();

    alert(myInstance.ancestor === MyModel);                               // true
    alert(myInstance.ancestor.ancestor === BaseModel);                    // true

    alert(myInstance.url());                                              // "/users/ID/somethingExtra"
