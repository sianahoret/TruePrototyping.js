With TruePrototyping itself

    var BaseModel = {                                                     // this object shoulkdn't be Function, as in examples above
        url: function(){
          return '/users/ID';
        }
    };

    var MyModel = BaseModel.derive({
      url: function(){
        return MyModel.super('url') + '/somethingExtra';                  // super is called immediately on MyModel, not on MyModel.prototype , as in examples above
        // return this.ancestor.super('url') + '/somethingExtra';         // this approach is also possible, but it is not good for myInstanceLevel2
      }
    });

    var myInstance = MyModel.derive();                                    // 'new' operator is not needed, as in examples above

    alert(myInstance.ancestor === MyModel);                               // true (MyModel is immediate ancestor, there is no any 'prototype')
    alert(myInstance.ancestor.ancestor === BaseModel);                    // true

    alert(myInstance.url());                                              // "/users/ID/somethingExtra"
